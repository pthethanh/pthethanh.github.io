Iceberg #

Iceberg use JDBC/Glue/Hive to store metadata, just what table/namespace is created and where the meta is stored in the storage backend.

The storage backend can be S3 or others, Iceberg will use them to storage metadata, snapshot...

Spark will handle writing the data, not iceberg, that's why we have to specify spark fs configuration in spark config so that spark can write the data.

ProcessRawEvent.scala

import org.apache.spark.sql.functions._
import org.apache.spark.sql.streaming.Trigger
import org.apache.spark.sql.{Dataset, SparkSession}

object ProcessRawEvent {
  def main(args: Array[String]): Unit = {
    val appName = sys.env.getOrElse("APP_NAME", "hello-spark-kafka-structured-streaming")
    val kafkaUser = System.getenv("KAFKA_USERNAME")
    val kafkaPassword = System.getenv("KAFKA_PASSWORD")
    val kafkaConf = Map[String, String] (
      "kafka.bootstrap.servers" -> sys.env.getOrElse("KAFKA_BOOTSTRAP_SERVERS", "kafka:9092"),
      "subscribe" -> sys.env.getOrElse("SUBSCRIBE", "spark-streaming-test"),
      "kafka.sasl.mechanism" -> sys.env.getOrElse("KAFKA_SASL_MECHANISM", "PLAIN"),
      "kafka.security.protocol" -> sys.env.getOrElse("KAFKA_SECURITY_PROTOCOL", "SASL_PLAINTEXT"),
      "kafka.sasl.jaas.config" -> sys.env.getOrElse("KAFKA_SASL_JAAS_CONFIG", s"""org.apache.kafka.common.security.plain.PlainLoginModule required username="$kafkaUser" password="$kafkaPassword";"""),
    )

    val spark: SparkSession = SparkSession.builder().appName(appName).getOrCreate()
    import spark.implicits._

    // read data
    val df = spark.readStream.format("kafka")
      .options(kafkaConf) // must set the options here. putting in the spark config doesn't work.
      .load()
    val ds = df.selectExpr("CAST(value AS STRING) as value_json")
      .select(from_json(col("value_json"), Event.schema).as[Event])

    // init sql
    val initSQL = sys.env.getOrElse("INIT_SQL", "")
    if (initSQL.nonEmpty) {
      initSQL.split(";").foreach(spark.sql)
    }
    // write data
    val processData = (batch: Dataset[Event], id: Long) => {
      //batch.persist()
      //write emergency data
      batch.filter($"kind" === "emergency")
        .write
        .mode(sys.env.getOrElse("WRITE_MODE", "append"))
        .format("iceberg")
        .option("path", sys.env.getOrElse("EMERGENCY_EVENT_TABLE", "local.db.emergency"))
        .save()

      // write raw data
      batch.write.format("iceberg")
        .mode(sys.env.getOrElse("WRITE_MODE", "append"))
        .option("path", sys.env.getOrElse("RAW_EVENT_TABLE", "local.db.events"))
        .save()
      //batch.unpersist()
    }
    ds.writeStream
      .option("checkpointLocation", sys.env.getOrElse("CHECKPOINT_LOCATION", ""))
      .outputMode(sys.env.getOrElse("OUTPUT_MODE", "append"))
      .foreachBatch(processData)
      .queryName("eventStream")
      .trigger(Trigger.ProcessingTime(sys.env.getOrElse("TRIGGER_PROCESSING_TIME", "1 seconds")))
      .start()
      .awaitTermination()
  }
}

Event.scala

import org.apache.spark.sql.types.{StringType, StructField, StructType, TimestampType}

import java.sql.Timestamp

case class Event(uuid: String, kind: String, data: String, ts: Timestamp)

object Event {
  val schema: StructType = StructType(Seq(
    StructField("uuid", StringType, nullable = false),
    StructField("kind", StringType, nullable = false),
    StructField("data", StringType, nullable = false),
    StructField("ts", TimestampType, nullable = false)
  ))
}

EventGenerator.scala


import com.fasterxml.jackson.module.scala.DefaultScalaModule
import com.fasterxml.jackson.databind.ObjectMapper

import java.sql.Timestamp
import java.time.Instant
import java.util.UUID

object EventGenerator {
  def main(args: Array[String]): Unit = {
    val mapper = new ObjectMapper()
    mapper.registerModule(DefaultScalaModule)

    for (i <- 1 to 10) {
      val e = Event(uuid = UUID.randomUUID().toString, kind="call", data = """{"value":20}""", ts = Timestamp.from(Instant.now()));
      val s = mapper.writeValueAsString(e)
      println(s)
    }
  }
}
